```csharp
// JWT配置
public class JwtOptions
{
    public string SecretKey { get; set; }        // 至少32字符
    public string Issuer { get; set; }           // 发行者
    public string Audience { get; set; }          // 受众
    public int ExpirationMinutes { get; set; }   // 过期时间
}

// Token生成
public class JwtTokenService
{
    public string GenerateToken(UserClaims claims)
    {
        var key = new SymmetricSecurityKey(Encoding.UTF8.GetBytes(_options.SecretKey));
        var credentials = new SigningCredentials(key, SecurityAlgorithms.HmacSha256);
        
        var token = new JwtSecurityToken(
            issuer: _options.Issuer,
            audience: _options.Audience,
            claims: new[]
            {
                new Claim(JwtRegisteredClaimNames.Sub, claims.UserId),
                new Claim(JwtRegisteredClaimNames.Jti, Guid.NewGuid().ToString()),
                new Claim("username", claims.Username),
                new Claim("role", claims.Role)
            },
            expires: DateTime.UtcNow.AddMinutes(_options.ExpirationMinutes),
            signingCredentials: credentials
        );
        
        return new JwtSecurityTokenHandler().WriteToken(token);
    }
}
```

## 3. 密码加密

### 3.1 AES-256-GCM加密

```csharp
/// <summary>
/// AES-256-GCM密码加密服务
/// </summary>
public class AesPasswordService : IPasswordService
{
    private readonly byte[] _key;
    private const int NonceSize = 12;  // 96-bit nonce
    private const int TagSize = 16;    // 128-bit tag
    
    public AesPasswordService(IOptions<EncryptionOptions> options)
    {
        // 密钥必须为32字节 (256-bit)
        _key = DeriveKey(options.Value.PasswordKey);
    }
    
    public string Encrypt(string plainText)
    {
        using var aes = new AesGcm(_key, TagSize);
        
        // 生成随机nonce
        var nonce = new byte[NonceSize];
        RandomNumberGenerator.Fill(nonce);
        
        // 加密
        var plainBytes = Encoding.UTF8.GetBytes(plainText);
        var cipherBytes = new byte[plainBytes.Length];
        var tag = new byte[TagSize];
        
        aes.Encrypt(nonce, plainBytes, cipherBytes, tag);
        
        // 组合: nonce + tag + ciphertext
        var result = new byte[nonce.Length + tag.Length + cipherBytes.Length];
        Buffer.BlockCopy(nonce, 0, result, 0, nonce.Length);
        Buffer.BlockCopy(tag, 0, result, nonce.Length, tag.Length);
        Buffer.BlockCopy(cipherBytes, 0, result, nonce.Length + tag.Length, cipherBytes.Length);
        
        return Convert.ToBase64String(result);
    }
    
    public string Decrypt(string cipherText)
    {
        var data = Convert.FromBase64String(cipherText);
        
        using var aes = new AesGcm(_key, TagSize);
        
        // 分解: nonce + tag + ciphertext
        var nonce = new byte[NonceSize];
        var tag = new byte[TagSize];
        var cipherBytes = new byte[data.Length - NonceSize - TagSize];
        
        Buffer.BlockCopy(data, 0, nonce, 0, NonceSize);
        Buffer.BlockCopy(data, NonceSize, tag, 0, TagSize);
        Buffer.BlockCopy(data, NonceSize + TagSize, cipherBytes, 0, cipherBytes.Length);
        
        // 解密
        var plainBytes = new byte[cipherBytes.Length];
        aes.Decrypt(nonce, cipherBytes, tag, plainBytes);
        
        return Encoding.UTF8.GetString(plainBytes);
    }
    
    private static byte[] DeriveKey(string password)
    {
        using var deriveBytes = new Rfc2898DeriveBytes(
            password, 
            salt: Encoding.UTF8.GetBytes("DongQiDB_Salt"), 
            iterations: 100000,
            HashAlgorithmName.SHA256);
        return deriveBytes.GetBytes(32);
    }
}
```

## 4. SQL安全

### 4.1 SQL验证器

```csharp
/// <summary>
/// SQL安全过滤器 - 防护SQL注入和恶意操作
/// </summary>
public class SqlSecurityFilter : ISqlSecurityFilter
{
    // 禁止的SQL关键词
    private static readonly HashSet<string> BlockedKeywords = new(StringComparer.OrdinalIgnoreCase)
    {
        // DDL (数据定义语言)
        "CREATE", "ALTER", "DROP", "TRUNCATE", "RENAME",
        
        // DML (数据操作语言)
        "INSERT", "UPDATE", "DELETE", "MERGE",
        
        // DCL (数据控制语言)
        "GRANT", "REVOKE", "DENY",
        
        // TCL (事务控制)
        "COMMIT", "ROLLBACK", "SAVEPOINT",
        
        // 存储过程和函数
        "EXEC", "EXECUTE", "CALL",
        
        // 系统命令
        "xp_", "sp_", "sys.", "pg_",
        
        // 其他危险操作
        "INTO OUTFILE", "INTO DUMPFILE", "LOAD_FILE",
        "INFORMATION_SCHEMA", "PROCESSLIST"
    };
    
    // 只允许的SQL类型
    private static readonly HashSet<string> AllowedSqlTypes = new(StringComparer.OrdinalIgnoreCase)
    {
        "SELECT", "SHOW", "DESCRIBE", "DESC", "EXPLAIN"
    };
    
    /// <summary>
    /// 验证SQL是否为只读查询
    /// </summary>
    public SqlValidationResult Validate(string sql, DatabaseType databaseType)
    {
        var result = new SqlValidationResult { IsValid = true };
        var normalizedSql = sql.Trim();
        
        // 1. 检查是否为空
        if (string.IsNullOrWhiteSpace(normalizedSql))
        {
            result.IsValid = false;
            result.Errors.Add("SQL语句不能为空");
            return result;
        }
        
        // 2. 提取SQL类型
        var sqlType = ExtractSqlType(normalizedSql);
        
        // 3. 检查是否为允许的SQL类型
        if (!AllowedSqlTypes.Contains(sqlType))
        {
            result.IsValid = false;
            result.Errors.Add($"仅允许SELECT查询，当前SQL类型为: {sqlType}");
            result.IsReadOnly = false;
            return result;
        }
        
        result.IsReadOnly = true;
        
        // 4. 检查禁止关键词
        foreach (var keyword in BlockedKeywords)
        {
            if (ContainsKeyword(normalizedSql, keyword))
            {
                result.IsValid = false;
                result.Errors.Add($"SQL包含禁止关键词: {keyword}");
            }
        }
        
        // 5. 检查嵌套查询深度
        var nestLevel = CalculateNestingLevel(normalizedSql);
        if (nestLevel > 5)
        {
            result.Warnings.Add($"嵌套查询深度为{nestLevel}层，建议简化");
        }
        
        // 6. 检查IN子句长度
        var inClauseLength = CheckInClauseLength(normalizedSql);
        if (inClauseLength > 1000)
        {
            result.Warnings.Add($"IN子句包含{inClauseLength}个元素，可能影响性能");
        }
        
        return result;
    }
    
    private static string ExtractSqlType(string sql)
    {
        var words = sql.Split(new[] { ' ', '\t', '\n', '\r' }, 
            StringSplitOptions.RemoveEmptyEntries);
        return words.FirstOrDefault()?.ToUpperInvariant() ?? "";
    }
    
    private static bool ContainsKeyword(string sql, string keyword)
    {
        var pattern = $@"\b{Regex.Escape(keyword)}\b";
        return Regex.IsMatch(sql, pattern, RegexOptions.IgnoreCase);
    }
}
```

## 5. 输入过滤

### 5.1 自然语言输入过滤

```csharp
/// <summary>
/// 用户输入干扰过滤
/// </summary>
public class InputFilterService : IInputFilterService
{
    // 干扰词列表
    private static readonly Dictionary<string, string[]> NoisePatterns = new()
    {
        // 口语化表达
        ["口语词"] = new[] { "那个", "嗯", "啊", "呃", "这个", "就是", "的话" },
        
        // 无意义填充
        ["填充词"] = new[] { "帮我", "请", "能不能", "是否可以", "麻烦", "谢谢" },
        
        // 干扰意图
        ["误导词"] = new[] { "顺便", "对了", "另外", "还有" }
    };
    
    // SQL保留字白名单
    private static readonly HashSet<string> SqlReservedWords = new(StringComparer.OrdinalIgnoreCase)
    {
        "SELECT", "FROM", "WHERE", "AND", "OR", "NOT", "IN", "LIKE",
        "BETWEEN", "IS", "NULL", "AS", "JOIN", "LEFT", "RIGHT", "INNER",
        "OUTER", "ON", "GROUP", "BY", "ORDER", "ASC", "DESC", "LIMIT",
        "OFFSET", "HAVING", "DISTINCT", "COUNT", "SUM", "AVG", "MAX", "MIN",
        "DATE", "TIME", "YEAR", "MONTH", "DAY", "HOUR", "MINUTE", "SECOND",
        "NOW", "TODAY", "YESTERDAY", "TOMORROW", "THIS", "LAST", "NEXT"
    };
    
    /// <summary>
    /// 过滤干扰数据
    /// </summary>
    public string FilterNoise(string input)
    {
        if (string.IsNullOrWhiteSpace(input))
            return string.Empty;
        
        var result = input.Trim();
        
        // 1. 移除干扰词
        foreach (var pattern in NoisePatterns.Values.SelectMany(p => p))
        {
            result = result.Replace(pattern, " ");
        }
        
        // 2. 规范化空格
        result = Regex.Replace(result, @"\s+", " ");
        
        // 3. 移除特殊字符 (保留中文、英文、数字、常用标点)
        result = Regex.Replace(result, @"[^\u4e00-\u9fa5a-zA-Z0-9\s,，。、:：\-（）()]", "");
        
        return result.Trim();
    }
    
    /// <summary>
    /// 提取查询意图
    /// </summary>
    public QueryIntent ExtractIntent(string input)
    {
        var filtered = FilterNoise(input);
        
        // 时间意图识别
        var timePattern = ExtractTimePattern(filtered);
        
        // 聚合意图识别
        var aggregationType = DetectAggregation(filtered);
        
        // 排序意图识别
        var sortOrder = DetectSortOrder(filtered);
        
        return new QueryIntent
        {
            OriginalInput = input,
            FilteredInput = filtered,
            TimePattern = timePattern,
            AggregationType = aggregationType,
            SortOrder = sortOrder
        };
    }
    
    /// <summary>
    /// 提取时间模式
    /// </summary>
    private TimePattern ExtractTimePattern(string input)
    {
        if (input.Contains("今天")) return TimePattern.Today;
        if (input.Contains("昨天")) return TimePattern.Yesterday;
        if (input.Contains("本周") || input.Contains("这周")) return TimePattern.ThisWeek;
        if (input.Contains("本月") || input.Contains("这个月")) return TimePattern.ThisMonth;
        if (input.Contains("今年") || input.Contains("这一年")) return TimePattern.ThisYear;
        
        return TimePattern.Unspecified;
    }
    
    /// <summary>
    /// 检测聚合类型
    /// </summary>
    private AggregationType DetectAggregation(string input)
    {
        if (input.Contains("多少") || input.Contains("总数") || input.Contains("计数"))
            return AggregationType.Count;
        if (input.Contains("总和") || input.Contains("合计") || input.Contains("总额"))
            return AggregationType.Sum;
        if (input.Contains("平均") || input.Contains("均值"))
            return AggregationType.Avg;
        if (input.Contains("最大") || input.Contains("最高") || input.Contains("最多"))
            return AggregationType.Max;
        if (input.Contains("最小") || input.Contains("最低") || input.Contains("最少"))
            return AggregationType.Min;
        
        return AggregationType.None;
    }
}
```

### 6.1 限流策略

```csharp
/// <summary>
/// 基于IP和用户的复合限流
/// </summary>
public class RateLimitingMiddleware
{
    private readonly Dictionary<string, RateLimitEntry> _entries = new();
    private readonly SemaphoreSlim _lock = new(1, 1);
    
    // 限流配置
    private static readonly Dictionary<string, RateLimitRule> Rules = new()
    {
        ["default"] = new RateLimitRule { PermitLimit = 100, Window = 60 },
        ["ai"] = new RateLimitRule { PermitLimit = 20, Window = 60 },
        ["export"] = new RateLimitRule { PermitLimit = 10, Window = 60 },
        ["auth"] = new RateLimitRule { PermitLimit = 5, Window = 60 }
    };
    
    public async Task<RateLimitResult> CheckRateLimitAsync(
        string key, 
        string endpoint)
    {
        var rule = GetRuleForEndpoint(endpoint);
        await _lock.WaitAsync();
        
        try
        {
            var now = DateTime.UtcNow;
            var windowStart = now.AddSeconds(-rule.Window);
            
            // 获取或创建条目
            if (!_entries.TryGetValue(key, out var entry))
            {
                entry = new RateLimitEntry();
                _entries[key] = entry;
            }
            
            // 清理过期记录
            entry.Requests.RemoveAll(t => t < windowStart);
            
            // 检查限制
            if (entry.Requests.Count >= rule.PermitLimit)
            {
                var retryAfter = rule.Window - (now - entry.Requests.Min()).TotalSeconds;
                return new RateLimitResult
                {
                    IsAllowed = false,
                    RetryAfterSeconds = (int)retryAfter
                };
            }
            
            // 记录请求
            entry.Requests.Add(now);
            
            return new RateLimitResult
            {
                IsAllowed = true,
                Remaining = rule.PermitLimit - entry.Requests.Count,
                Limit = rule.PermitLimit
            };
        }
        finally
        {
            _lock.Release();
        }
    }
}
```
