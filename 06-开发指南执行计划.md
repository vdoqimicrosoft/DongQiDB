```

### 4.2 Commit规范

```

### 4.3 Pull Request流程

```
1. 创建分支
   git checkout -b feature/T00-001-user-authentication

2. 提交代码
   git add .
   git commit -m "feat(T00-001): 实现用户认证功能"

3. 推送分支
   git push -u origin feature/T00-001-user-authentication

4. 创建PR
   标题: [T00-001] 用户认证功能实现
   内容: 参考PR模板

5. 代码审查
   - 至少1人审查
   - 所有检查通过
   - 无冲突

6. 合并
   Squash and merge
```

### 5.3 测试示例

```csharp
// Unit Test Example
public class TextToSqlServiceTests
{
    private readonly Mock<IAiClient> _aiClientMock;
    private readonly TextToSqlService _service;

    public TextToSqlServiceTests()
    {
        _aiClientMock = new Mock<IAiClient>();
        _service = new TextToSqlService(_aiClientMock.Object);
    }

    [Fact]
    public async Task TextToSqlAsync_ValidInput_ReturnsSql()
    {
        // Arrange
        var request = new TextToSqlRequest
        {
            NaturalLanguage = "查询所有用户",
            SchemaContext = "users(id, name, email)"
        };
        
        _aiClientMock
            .Setup(x => x.GenerateAsync(It.IsAny<string>()))
            .ReturnsAsync("SELECT * FROM users");

        // Act
        var result = await _service.TextToSqlAsync(request);

        // Assert
        Assert.True(result.Success);
        Assert.Equal("SELECT * FROM users", result.Sql);
    }
}
```
